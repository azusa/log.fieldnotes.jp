---
title: "Joplin CLIでノートのアーカイブをimportしてmd形式でexport"
date: 2025-10-22
---

```
NPM_CONFIG_PREFIX=~/.joplin-bin npm install -g joplin
sudo ln -s ~/.joplin-bin/bin/joplin /usr/local/bin/joplin

mkdir joplin
cd joplin

sudo apt search 7zip
sudo apt install 7zip

rehash

# アーカイブを展開
p7zip -d 202510211738.7z

# インポート
joplin import all_notebooks.jex --format jex

# エクスポート
mkdir export
cd export
joplin export . --format md

# ストレージに同期
rsync -avz export/ /mnt/c/Users/azusa/Dropbox/notes/
```
