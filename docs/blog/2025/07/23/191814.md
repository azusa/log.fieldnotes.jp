title: "ラズパイの動画をYouTubeとMiniDLNAの両方に送る"
date: 2025-07-23
---

起動スクリプト

```
#!/bin/bash

# ここはnginxで公開している想定
HLS_DIR="/var/www/html/hls"

# systemdから起動するときにOS起動直後だとa.rtmp.youtube.comの名前解決に失敗するので入れている(なおしたい)
sleep 10

# cronからのバッチで温度と現在時刻の情報を書き出している
ffmpeg -stream_loop -1 -i /home/azusa/work/no_sound-1.mp3  -f v4l2 -s 1280x768 -r 10 -i /dev/video0 -vcodec libx264 -pix_fmt yuv420p -preset ultrafast -r 15 -g 30 -b:v 750k -acodec aac -ar 44100 -threads 0 -b:a 128k -bufsize 512k -vf drawtext="fontfile=/usr/share/fonts/truetype/noto/NotoMono-Regular.ttf:fontcolor=white:box=1:boxcolor=black:fontsize=24:textfile='/home/azusa/src/temper/text.txt':reload=1" -f tee -map 0:a -map 1:v "[f=flv:onfail=ignore]rtmp://a.rtmp.youtube.com/live2/xxxx-xxxx-xxxx-xxxx-xxxx[f=hls:hls_time=4:hls_list_size=5:hls_flags=delete_segments:onfail=ignore]${HLS_DIR}/stream.m3u8"
```

/etc/minidlna.conf(抜粋)

```
media_dir=V,/var/www/html/hls

inofity=yes

notify_interval_30
```
